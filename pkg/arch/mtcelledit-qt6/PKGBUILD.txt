pkgname=mtcelledit-qt6
pkgver
pkgrel=@MT_RELVER@
pkgdesc="Qt6 Spreadsheet GUI"
arch=('i686' 'x86_64' 'arm')
license=('GPL3')
depends=('qt6-base' 'libmtkit' 'libmtcelledit' 'libmtpixy' 'libmtcedui' 'libmtqex6')
install=mtcelledit_desktop.install
source=($pkgname-$pkgver.tar.gz)
md5sums

build() {
  cd "$srcdir/$pkgname-$pkgver"

  @MT_CONF@ --use-qt6
  make
}

package() {
  cd "$srcdir/$pkgname-$pkgver"

  make DESTDIR="$pkgdir/" install
}
