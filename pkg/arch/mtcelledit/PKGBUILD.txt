pkgname=mtcelledit
pkgver
pkgrel=@MT_RELVER@
pkgdesc="GTK+2 Spreadsheet GUI"
arch=('i686' 'x86_64' 'arm')
license=('GPL3')
depends=('gtk2' 'libmtkit' 'libmtcelledit' 'libmtcedui' 'libmtgex')
install=mtcelledit_desktop.install
source=($pkgname-$pkgver.tar.gz)
md5sums

build() {
  cd "$srcdir/$pkgname-$pkgver"

  @MT_CONF@ --enable-gtk2
  make
}

package() {
  cd "$srcdir/$pkgname-$pkgver"

  make DESTDIR="$pkgdir/" install
}
